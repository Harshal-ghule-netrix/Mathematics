interface Maths
{
	int add(int,int);
	int sub(int,int);
}

class Arithmetic : Maths
{

	public int sub(int a,int b){
	int sum=a-b;
	return sum;
	}

	int add(int iNo1 , int iNo2)
	{
		return iNo1 + iNo2;
	}
	
	int add(int iNo1 , int iNo2)
	{
		int iSum = iNo1 + iNo2;
		return iSum;
	}	
	int int multiply(int num1,int num2){
	int mul=num1*num2;
	return mul;
	}
}
